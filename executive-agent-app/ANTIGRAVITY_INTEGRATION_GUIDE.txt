class AgentLoop:
    def __init__(self, runtime):
        self.runtime = runtime
        self.commander = runtime["commander"]
        
    async def run(self, goal):
        # 1. Commander analyzes goal
        analysis = self.commander.analyze_goal(goal)
        
        # 2. Create execution plan
        plan = self.commander.plan_steps(goal)
        
        # 3. Execute each step
        results = []
        for step in plan["steps"]:
            agent_name = step["agent"]
            action = step["action"]
            
            # Route to correct agent
            result = await self.route_to_agent(agent_name, action, context)
            results.append(result)
            
            # Check result, retry if needed
            if not result.get("success"):
                result = await self.retry_step(step, result)
                
            # Log to workflow if recording
            self.log_step_to_workflow(step, result)
            
        return {"success": True, "results": results}

import asyncio

class AgentExecutor:
    async def execute_tool(self, tool_name, params):
        tool = self.registry.get_tool(tool_name)
        
        if asyncio.iscoroutinefunction(tool):
            # Already async
            return await tool(**params)
        else:
            # Wrap sync call
            return await asyncio.to_thread(tool, **params)

class AgentExecutor:
    async def execute_with_retry(self, agent, action, context, max_retries=3):
        for attempt in range(max_retries):
            try:
                result = await agent.handle_task(action, context)
                
                if result.get("success"):
                    return result
                    
                # Check if retryable
                error = result.get("error", "")
                if self.is_retryable_error(error):
                    wait_time = 2 ** attempt  # Exponential backoff
                    await asyncio.sleep(wait_time)
                    continue
                else:
                    return result
                    
            except Exception as e:
                if attempt == max_retries - 1:
                    return {"success": False, "error": str(e)}
                    
    def is_retryable_error(self, error):
        retryable_patterns = [
            "timeout", "connection", "element not found",
            "stale element", "page crash"
        ]
        return any(pattern in error.lower() for pattern in retryable_patterns)

class AgentLoop:
    def __init__(self, runtime):
        self.runtime = runtime
        self.workflow_tools = runtime["workflow_tools"]
        
    async def execute_step(self, step, context):
        # Before execution
        self.before_step_hook(step)
        
        # Execute
        result = await self.route_to_agent(step["agent"], step["action"], context)
        
        # After execution
        self.after_step_hook(step, result)
        
        # Auto-save successful workflows
        if result.get("success") and self.is_reusable_sequence():
            self.prompt_save_workflow()
            
        return result
        
    def before_step_hook(self, step):
        if self.context.workflow_recording:
            self.pending_step = {
                "action": step["action"],
                "params": step.get("params", {}),
                "start_time": time.time()
            }
            
    def after_step_hook(self, step, result):
        if self.context.workflow_recording and hasattr(self, 'pending_step'):
            self.context.record_step(
                self.pending_step["action"],
                self.pending_step["params"],
                result
            )
            del self.pending_step

class PermissionManager:
    def __init__(self, permissions_file="config/permissions.yaml"):
        import yaml
        with open(permissions_file) as f:
            self.permissions = yaml.safe_load(f)
            
    def check_permission(self, action, context=None):
        action_config = self.permissions["permissions"].get(action, {})
        setting = action_config.get("default", "ask")
        
        if setting == "allow":
            return True
        elif setting == "deny":
            return False
        elif setting == "ask":
            return self.prompt_user(action, context)
            
    def prompt_user(self, action, context):
        # CLI version
        response = input(f"Allow {action}? (y/n): ")
        return response.lower() == "y"

class MainWindow:
    def __init__(self):
        # ... existing setup ...
        
        # Connect to backend
        from src.main import build_runtime
        self.runtime = build_runtime()
        self.commander = self.runtime["commander"]
        
    def send_message(self):
        message = self.message_input.text().strip()
        if not message:
            return
            
        # Add to chat
        self.chat_history.add_user_message(message)
        self.message_input.clear()
        
        # Run in background thread
        self.worker = AgentWorker(self.commander, message)
        self.worker.finished.connect(self.on_agent_finished)
        self.worker.progress.connect(self.on_agent_progress)
        self.worker.start()
        
    def on_agent_finished(self, result):
        if result.get("success"):
            self.chat_history.add_agent_message(
                result.get("summary", "Task completed successfully")
            )
        else:
            self.chat_history.add_system_message(
                f"Error: {result.get('error', 'Unknown error')}"
            )
            
    def on_agent_progress(self, message):
        self.status_bar.showMessage(message)
